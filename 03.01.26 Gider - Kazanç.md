#### 2. **CDN Edge Caching**

 

```javascript

// Cloudflare Workers (Edge computing)

export default {

  async fetch(request, env) {

    const cache = caches.default;

    const cacheKey = new Request(request.url, request);

 

    // Check cache first

    let response = await cache.match(cacheKey);

 

    if (!response) {

      // Cache miss - fetch from origin

      response = await fetch(request);

 

      // Cache for 1 hour

      const headers = new Headers(response.headers);

      headers.set('Cache-Control', 'public, max-age=3600');

 

      response = new Response(response.body, {

        status: response.status,

        headers: headers

      });

 

      await cache.put(cacheKey, response.clone());

    }

 

    return response;

  }

}

```

 

#### 3. **Redis Caching Layer**

 

```javascript

const Redis = require('ioredis');

// Feed endpoint with cache

app.get('/feed', async (req, res) => {

    const cacheKey = `feed:${userId}:${page}`;

 

    // Check cache (1ms)

    let videos = await redis.get(cacheKey);

 

    if (videos) {

        return res.json(JSON.parse(videos));

    }

 

    // Cache miss - fetch from Supabase (50ms)

    videos = await supabase

        .from('videos')

        .select('*')

        .range(page * 10, (page + 1) * 10);

 

    // Cache for 5 minutes

    await redis.setex(cacheKey, 300, JSON.stringify(videos));

 

    res.json(videos);

});

```

#### 1. Recommendation Algorithm (For You Page)

 

```sql

CREATE OR REPLACE FUNCTION get_recommended_videos(user_id UUID)

RETURNS TABLE (video_id UUID, score FLOAT) AS $$

BEGIN

  RETURN QUERY

  WITH user_interactions AS (

    SELECT video_id, COUNT(*) as engagement

    FROM user_likes WHERE user_id = user_id

  ),

  similar_users AS (

    SELECT DISTINCT l2.user_id

    FROM user_likes l1

    JOIN user_likes l2 ON l1.video_id = l2.video_id

    WHERE l1.user_id = user_id AND l2.user_id != user_id

  )

  SELECT v.id,

         (0.4 * COALESCE(trending_score, 0) +

          0.3 * COALESCE(engagement_score, 0) +

          0.3 * COALESCE(similarity_score, 0)) as score

  FROM videos v

  ORDER BY score DESC

  LIMIT 20;

END;

$$ LANGUAGE plpgsql;

```
